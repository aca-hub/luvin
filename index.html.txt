<!-- Cloudinary Upload Widget Script (Free Service) -->
<script src="upload-widget.cloudinary.com" type="text/javascript"></script>

<script type="text/javascript">  
    // Replace 'your_cloud_name' and 'your_preset' with your actual Cloudinary details
    const myWidget = cloudinary.createUploadWidget({
        cloudName: 'demo', 
        uploadPreset: 'unsigned_preset'
    }, (error, result) => { 
        if (!error && result && result.event === "success") { 
            console.log('Done! Here is the image info: ', result.info); 
            addMediaToGallery(result.info);
        }
    });

    document.getElementById("upload_widget").addEventListener("click", function(){
        myWidget.open();
    }, false);

    function addMediaToGallery(info) {
        const gallery = document.getElementById('media-gallery');
        const div = document.createElement('div');
        div.className = 'media-item';
        
        if (info.resource_type === "image") {
            div.innerHTML = `<img src="${info.secure_url}" alt="Uploaded Image">`;
        } else {
            div.innerHTML = `<video controls><source src="${info.secure_url}"></video>`;
        }
        gallery.prepend(div);
    }
</script>
